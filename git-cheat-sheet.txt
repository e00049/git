
git branch   -  Active branch - HEAD

git checkout development  - switch to exitsting branch

git checkout -b uat  - change create 

git merge uat  - uat will be merge with main branch

git branch -d uat

git mv old-file-name new-file-name 


git reset file-name   - Remove from staging area to normal area

git reset --soft HEAD^  -  Undo last commit 

git show HEAD~1:file.txt   -  How was the file while commit 1 from top to down 
git show HEAD~2:file.txt > stage-002.txt  - what was the content at log2 stage

git diff HEAD~2 file.txt   - difference between current and from 2nd commit

git restore -s HEAD~1 file.txt  - move to log 01 stage
