git config --global user.email "you@example.com"   
git config --global user.name "Your Name"                  - this is for entire system 
git config --global core.editor "/usr/bin/nano"     
git remote add origin https://github.com/e00049/git.git    - add your remote repo
git remote -v           -> list of bookmarked repository names and additionally, the corresponding repository URL.

git init                    - Git Init on your working directory
git pull origin main        - We could use git pull before push into remote repo
git add file1 file2         - Local to staging Area
git stage file1 file2       - Local to staging Area
git commit -m "commit msg"  - Staging area to Local repo
git push origin main        - Local repo to your remote repo

git config --edit                    - To edit the config files
git remote set-url origin <SSH_URL>  - To edit https to ssh url
